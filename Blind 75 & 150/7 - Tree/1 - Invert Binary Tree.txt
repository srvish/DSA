You are given the root of a binary tree root. Invert the binary tree and return its root.

Example 1:
Input: root = [1,2,3,4,5,6,7]
Output: [1,3,2,7,6,5,4]

Example 2:
Input: root = [3,2,1]
Output: [3,1,2]

Example 3:
Input: root = []
Output: []

Constraints:
    0 <= The number of nodes in the tree <= 100.
    -100 <= Node.val <= 100

Solution:
/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     public int val;
 *     public TreeNode left;
 *     public TreeNode right;
 *     public TreeNode(int val=0, TreeNode left=null, TreeNode right=null) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */

public class Solution {
    public TreeNode InvertTree(TreeNode root) {
        return AnotherStackApproach(root);
    }

    private TreeNode RecursiveApproach(TreeNode root) {
        if (root == null) {
            return root;
        }

        TreeNode temp = root.left;
        root.left = root.right;
        root.right = temp;

        InvertTree(root.left);
        InvertTree(root.right);

        return root;
    }

    private TreeNode AnotherStackApproach(TreeNode node) {
        if (node == null) {
            return node;
        }

        Stack<TreeNode> st = new Stack<TreeNode>();
        st.Push(node);
        while(st.Count > 0) {
            var temp = st.Pop();
            var tempNode = temp.left;
            temp.left = temp.right;
            temp.right = tempNode;
            if (temp.left != null) {
                st.Push(temp.left);
            }
            if (temp.right != null) {
                st.Push(temp.right);
            }
        }

        return node;
    }
}
